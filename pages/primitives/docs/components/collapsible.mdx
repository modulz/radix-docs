### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-collapsible
```

Then import the component in your component file.

```jsx
// Avatar.tsx
import * as CollapsiblePrimitive from '@interop-ui/react-collapsible';
```

### Basic example

```jsx live manual
// Collapsible.tsx
// import * as CollapsiblePrimitive from '@interop-ui/react-collapsible';
// import { Button } from '@radix-ui/ds-button';

render(
  <CollapsiblePrimitive.Collapsible>
    <CollapsiblePrimitive.CollapsibleButton as={Button}>
      Button
    </CollapsiblePrimitive.CollapsibleButton>
    <CollapsiblePrimitive.CollapsibleContent>
      Content 1
    </CollapsiblePrimitive.CollapsibleContent>
  </CollapsiblePrimitive.Collapsible>
);
```
