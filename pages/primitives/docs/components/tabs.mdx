description: An interface of layered content where activated by tab buttons.
features:
  - Full keyboard navigation.
  - Can be controlled or uncontrolled.

### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-tabs
```

Then import the component in your component file.

```jsx
// Tabs.tsx
import * as TabsPrimitive from '@radix-ui/react-tabs';
```

### Basic example

Create your styled tabs component from the tabs primitives.

```jsx live manual
// Tabs.tsx
// import * as TabsPrimitive from '@radix-ui/react-tabs';

const Tabs = styled(TabsPrimitive, {
  width: '20rem',
  maxWidth: '100%',
});

const TabsList = styled(TabsPrimitive.List, {
  flexShrink: 0,
  display: 'flex',
  '&[data-orientation="horizontal"]': {
    flexDirection: 'row',
    borderBottom: '1px solid $gray500',
  },
  '&[data-orientation="vertical"]': {
    flexDirection: 'column',
    borderRight: '1px solid $gray500',
  },
});

const TabsTab = styled(TabsPrimitive.Tab, {
  flexShrink: 0,
  padding: '0.4em 0.6em',
  fontWeight: '500',
  lineHeight: '1',
  userSelect: 'none',

  '&[data-orientation="horizontal"]': {
    borderBottom: '4px solid var(--border-color, transparent)',
  },

  '&[data-orientation="vertical"]': {
    borderRight: '4px solid var(--border-color, transparent)',
  },

  '&:focus': {
    '--border-color': '$black',
    outline: '1px solid var(--border-color)',
  },

  '&[data-disabled]': {
    color: '$gray300',
  },

  '&[data-state="active"]': {
    '--border-color': '$red500',
  },
});

const TabsPanel = styled(TabsPrimitive.Panel, {
  flexGrow: 1,
  padding: '1em',
  border: '1px solid $gray200',
});

render(
  <Tabs defaultSelectedId="tab1">
    <TabsList aria-label="tabs example">
      <TabsTab id="tab1">Tab 1</TabsTab>
      <TabsTab id="tab2" disabled>
        Tab 2
      </TabsTab>
      <TabsTab id="tab3">Tab 3</TabsTab>
    </TabsList>
    <TabsPanel id="tab1">
      Dis metus rhoncus sit convallis sollicitudin vel cum, hac purus tincidunt
      eros sem himenaeos integer, faucibus varius nullam nostra bibendum
      consectetur mollis, gravida elementum pellentesque volutpat dictum ipsum.
    </TabsPanel>
    <TabsPanel id="tab2">You'll never find me!</TabsPanel>
    <TabsPanel id="tab3">
      Ut nisi elementum metus semper mauris dui fames accumsan aenean, maecenas
      ac sociis dolor quam tempus pretium.
    </TabsPanel>
  </Tabs>
);
```

#### TabsList

`TabsList` displays the `TabsTab` components aligned along the edge of the active `TabsPanel`.
