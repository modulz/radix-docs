version: 0.0.1-13
description: An interface of layered content where activated by tab buttons.
features:
  - Can be controlled or uncontrolled.
  - Supports horizontal/vertical orientation.
  - Supports automatic/manual activation.
  - Full keyboard navigation.

### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-tabs
```

Then import the component in your component file.

```jsx
// Tabs.tsx
import * as TabsPrimitive from '@radix-ui/react-tabs';
```

### Basic example

Create your styled tabs component from the tabs primitives.

```jsx live manual
// Tabs.tsx
// import * as TabsPrimitive from '@radix-ui/react-tabs';

const Tabs = styled(TabsPrimitive.Root, {
  display: 'flex',
  flexDirection: 'column',
});

const TabsList = styled(TabsPrimitive.List, {
  flexShrink: 0,
  display: 'flex',
  borderBottom: '1px solid gainsboro',
});

const TabsTab = styled(TabsPrimitive.Tab, {
  flexShrink: 0,
  padding: '10px 20px',
  color: 'slategray',
  userSelect: 'none',

  '&:hover': { color: 'black' },

  '&[data-state="active"]': {
    color: 'black',
    boxShadow: 'inset 0 -1px 0 0 currentColor, 0 1px 0 0 currentColor',
  },
});

const TabsPanel = styled(TabsPrimitive.Panel, {
  flexGrow: 1,
  padding: 20,
});

render(
  <Tabs defaultSelectedId="tab1">
    <TabsList aria-label="tabs example">
      <TabsTab id="tab1">One</TabsTab>
      <TabsTab id="tab2">Two</TabsTab>
      <TabsTab id="tab3">Three</TabsTab>
    </TabsList>
    <TabsPanel id="tab1">Tab one content</TabsPanel>
    <TabsPanel id="tab2">Tab two content</TabsPanel>
    <TabsPanel id="tab3">Tab three content</TabsPanel>
  </Tabs>
);
```
