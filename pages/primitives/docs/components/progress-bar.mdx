### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-progress-bar
```

Then import the component in your component file.

```jsx
// ProgressBar.tsx
import * as ProgressBarPrimitive from '@radix-ui/react-progress-bar';
```

### Basic example

```jsx live manual
// ProgressBar.tsx
// import * as ProgressBarPrimitive from '@radix-ui/react-progress-bar';

const ProgressBar = styled(ProgressBarPrimitive.Root, {
  position: 'relative',
  height: 10,
  overflow: 'hidden',
  borderRadius: 5,
  background: 'gainsboro',
});

const ProgressBarIndicator = styled(ProgressBarPrimitive.Indicator, {
  boxSizing: 'border-box',
  position: 'absolute',
  backgroundColor: 'dodgerblue',
  height: '100%',
});

render(
  <ProgressBar value={50}>
    <ProgressBarIndicator style={{ width: '50%' }} />
  </ProgressBar>
);
```
