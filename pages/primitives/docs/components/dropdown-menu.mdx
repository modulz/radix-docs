---
title: Dropdown Menu
name: dropdown-menu
version: 0.0.1-1
description: Useful for displaying a list of contextual actions when clicking a trigger.
features:
  - Can be controlled or uncontrolled.
  - Supports items, labels, groups of items.
  - Supports checkable items (single or multiple).
  - Customise side, alignment, offsets, collision handling.
  - Optionally render a pointing arrow.
  - Focus is fully managed.
  - Full keyboard navigation.
  - Dismissing and layering behavior is highly customisable.
---

### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-dropdown-menu
```

Then import the component in your component file.

```jsx
// DropdownMenu.tsx
import * as DropdownMenuPrimitive from '@radix-ui/react-dropdown-menu';
```

### Basic example

```jsx live manual
// DropdownMenu.tsx
// import * as DropdownMenuPrimitive from '@radix-ui/react-dropdown-menu';
// import { Button } from '@radix-ui/ds-button';

const MenuPopper = styled(DropdownMenuPrimitive.DropdownMenuPopper, {
  minWidth: 130,
  backgroundColor: 'white',
  borderRadius: 6,
  padding: 5,
  boxShadow:
    'hsla(206,22%,7%,.15) 0px 5px 15px -5px, hsla(206,22%,7%,.1) 0px 5px 10px -15px',
});

const MenuItem = styled(DropdownMenuPrimitive.DropdownMenuItem, {
  fontSize: 13,
  padding: '5px 10px',
  borderRadius: 3,
  cursor: 'default',

  '&[data-disabled]': { color: 'slategray' },

  ':focus': {
    outline: 'none',
    backgroundColor: 'dodgerblue',
    color: 'white',
  },
});

const MenuSeparator = styled(DropdownMenuPrimitive.DropdownMenuSeparator, {
  height: 1,
  backgroundColor: 'gainsboro',
  margin: '5px 10px',
});

const MenuArrow = styled(DropdownMenuPrimitive.DropdownMenuArrow, {
  fill: 'white',
});

render(
  <DropdownMenuPrimitive.Root>
    <DropdownMenuPrimitive.DropdownMenuTrigger as={Button}>
      Open
    </DropdownMenuPrimitive.DropdownMenuTrigger>
    <MenuPopper sideOffset={5}>
      <MenuItem onSelect={() => alert('undo')}>Undo</MenuItem>
      <MenuItem onSelect={() => alert('redo')}>Redo</MenuItem>
      <MenuSeparator />
      <MenuItem disabled onSelect={() => alert('cut')}>
        Cut
      </MenuItem>
      <MenuItem onSelect={() => alert('copy')}>Copy</MenuItem>
      <MenuItem onSelect={() => alert('paste')}>Paste</MenuItem>
      <MenuArrow />
    </MenuPopper>
  </DropdownMenuPrimitive.Root>
);
```
