### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-avatar
```

Then import the component in your component file.

```jsx
// Avatar.tsx
import * as AvatarPrimitive from '@interop-ui/react-avatar';
```

### Basic example

```jsx live manual
// Avatar.tsx
// import * as AvatarPrimitive from '@interop-ui/react-avatar';

const Avatar = styled(AvatarPrimitive, {
  display: 'inline-flex',
  alignItems: 'center',
  justifyContent: 'center',
  verticalAlign: 'middle',
  overflow: 'hidden',
  userSelect: 'none',
  width: 48,
  height: 48,
  borderRadius: 24,
});

const AvatarImage = styled(AvatarPrimitive.Image, {
  width: '100%',
  height: '100%',
  objectFit: 'cover',
});

const AvatarFallback = styled(AvatarPrimitive.Fallback, {
  width: '100%',
  height: '100%',
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',
  backgroundColor: 'dodgerblue',
});

render(
  <Avatar>
    <AvatarImage src="https://picsum.photos/id/1005/400/400" />
    <AvatarFallback>UI</AvatarFallback>
  </Avatar>
);
```

#### Clickable Avatar with tooltip

```jsx live manual compact
// Avatar.tsx
// import * as AvatarPrimitive from '@interop-ui/react-avatar';
// import * as TooltipPrimitive from '@interop-ui/react-tooltip';

const Avatar = styled(AvatarPrimitive, {
  display: 'inline-flex',
  alignItems: 'center',
  justifyContent: 'center',
  verticalAlign: 'middle',
  overflow: 'hidden',
  userSelect: 'none',
  width: 48,
  height: 48,
  borderRadius: 24,
  appearance: 'none',
  border: 'none',
  padding: 0,
});

const AvatarImage = styled(AvatarPrimitive.Image, {
  width: '100%',
  height: '100%',
  objectFit: 'cover',
});

const AvatarFallback = styled(AvatarPrimitive.Fallback, {
  width: '100%',
  height: '100%',
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',
  backgroundColor: 'dodgerblue',
});

const TooltipTrigger = styled(TooltipPrimitive.Trigger, {
  backgroundColor: 'transparent',
  border: 'none',
  padding: 0,
});

const TooltipContent = styled(TooltipPrimitive.Content, {
  borderRadius: 3,
  padding: '3px 6px',
  fontSize: 12,
  backgroundColor: 'gainsboro',
  color: 'black',
});

render(
  <TooltipPrimitive>
    <TooltipTrigger>
      <Avatar onClick={() => alert('Clicked!')}>
        <AvatarImage src="https://picsum.photos/id/1006/400/400" />
        <AvatarFallback>AB</AvatarFallback>
      </Avatar>
    </TooltipTrigger>

    <TooltipPrimitive.Popper side="top">
      <TooltipContent>Tooltip content</TooltipContent>
      <TooltipPrimitive.Arrow style={{ fill: 'gainsboro' }} />
    </TooltipPrimitive.Popper>
  </TooltipPrimitive>
);
```
