title: Tooltip
name: tooltip
version: 0.0.1-13
description: An accessible Tooltip component that shows information related to an element when it is hovered or focused.
features:
  - Built in time machine to control appearance delays
---

### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-tooltip
```

Then import the component in your component file.

```jsx
// Tooltip.tsx
import * as TooltipPrimitive from '@radix-ui/react-tooltip';
```

### Basic example

```jsx live manual
// Tooltip.tsx
// import * as TooltipPrimitive from '@radix-ui/react-tooltip';
// import { IconButton } from '@radix-ui/ds-button'
// import { BlendingModeIcon } from '@modulz/radix-icons'

const TooltipContent = styled(TooltipPrimitive.Content, {
  borderRadius: 3,
  padding: '5px 10px',
  fontSize: 14,
  backgroundColor: 'gainsboro',
  color: 'black',
});

render(
  <TooltipPrimitive.Root>
    <TooltipPrimitive.Trigger as={IconButton}>
      <BlendingModeIcon aria-label="Blending mode" />
    </TooltipPrimitive.Trigger>
    <TooltipContent>
      Blending mode
      <TooltipPrimitive.Arrow style={{ fill: 'gainsboro' }} />
    </TooltipContent>
  </TooltipPrimitive.Root>
);
```

#### TooltipTrigger

The button that triggers the `Tooltip`. It is also used as the anchor the `TooltipContent` is positioned against.
