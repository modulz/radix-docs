### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-slider
```

Then import the component in your component file.

```jsx
// Slider.tsx
import * as SliderPrimitive from '@interop-ui/react-slider';
```

### Basic example

Create your styled slider components, then pass those styled components into `Slider` via the `as` prop:

```jsx live manual
// Slider.tsx
// import * as SliderPrimitive from '@interop-ui/react-slider';

// outer container styles
const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  flexShrink: 0,
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
});

// track styles
const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

// range styles
const SliderRange = styled(SliderPrimitive.Range, {
  backgroundColor: 'dodgerblue',
  position: 'absolute',
  borderRadius: '9999px',
  height: '100%',
});

// thumb styles
const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  ':focus': {
    outline: 'none',
    borderColor: 'blue',
  },
});

render(
  <Slider defaultValue={50}>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>
    <SliderThumb />
  </Slider>
);
```

### Examples

#### Vertical orientation

Create your styled slider components, then pass those styled components into `Slider` via the `as` prop:

```jsx live manual compact
// Slider.tsx
// import * as SliderPrimitive from '@interop-ui/react-slider';

// outer container styles
const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  flexShrink: 0,
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
  '&[data-orientation="vertical"]': {
    flexDirection: 'column',
    width: 16,
    height: 100,
  },
});

// track styles
const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
  '&[data-orientation="horizontal"]': {
    height: 2,
  },
  '&[data-orientation="vertical"]': {
    width: 2,
  },
});

// range styles
const SliderRange = styled(SliderPrimitive.Range, {
  backgroundColor: 'dodgerblue',
  position: 'absolute',
  borderRadius: '9999px',
  height: '100%',
  '&[data-orientation="horizontal"]': {
    height: '100%',
  },
  '&[data-orientation="vertical"]': {
    width: '100%',
  },
});

// thumb styles

  display: 'block',
```

#### Show thumb on hover

Create your styled slider components, then pass those styled components into `Slider` via the `as` prop:

```jsx live manual compact
// track styles
const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

// range styles
const SliderRange = styled(SliderPrimitive.Range, {
  backgroundColor: 'gray',
  position: 'absolute',
  borderRadius: '9999px',
  height: '100%',
});

// thumb styles
const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  opacity: 0,
  ':focus': {
    outline: 'none',
    borderColor: 'blue',
  },
});

// outer container styles
const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  flexShrink: 0,
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
  ':hover': {
    [`& ${SliderRange}`]: {
      backgroundColor: 'dodgerblue',
    },
    [`& ${SliderThumb}`]: {
      opacity: '1',
    },
  },
});

render(
  <Slider defaultValue={10}>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>
    <SliderThumb />
  </Slider>
);
```

#### Controlled state

The component will work as expected without any additional props or configuration, but if you need access to its state you can control it with the value and onChange props.

```jsx live manual compact
// Slider.tsx
// import * as SliderPrimitive from '@interop-ui/react-slider';

// outer container styles
const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  flexShrink: 0,
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
});

// track styles
const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

// range styles
const SliderRange = styled(SliderPrimitive.Range, {
  backgroundColor: 'dodgerblue',
  position: 'absolute',
  borderRadius: '9999px',
  height: '100%',
});

// thumb styles
const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  ':focus': {
    outline: 'none',
    borderColor: 'blue',
  },
});

function ControlledSlider() {
  const [value, setValue] = React.useState(10);
  return (
    <Slider value={value} onChange={(newValue) => setValue(newValue)}>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>
      <SliderThumb />
    </Slider>
  );
}

render(<ControlledSlider />);
```

#### SliderTrack

`SliderTrack` is the track part.

#### Range

`SliderRange` is the range part. Must live inside `SliderTrack`.

#### Thumb

`SliderThumb` is the thumb part. You can render multiple thumbs.
