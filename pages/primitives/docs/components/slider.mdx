### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-slider
```

Then import the component in your component file.

```jsx
// Slider.tsx
import * as SliderPrimitive from '@radix-ui/react-slider';
```

### Basic example

Create your styled slider component from the slider primitives.

```jsx live manual
// Slider.tsx
// import * as SliderPrimitive from '@radix-ui/react-slider';

const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
});

const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

const SliderRange = styled(SliderPrimitive.Range, {
  position: 'absolute',
  backgroundColor: 'dodgerblue',
  borderRadius: '9999px',
  height: '100%',
});

const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  ':focus': {
    outline: 'none',
    borderColor: 'dodgerblue',
  },
});

render(
  <Slider defaultValue={50}>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>
    <SliderThumb />
  </Slider>
);
```

  position: 'absolute',
  backgroundColor: 'dodgerblue',
  borderRadius: '9999px',
  height: '100%',
});

const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,

#### Disabled style

Use the availabe `data-*` attribute selectors to style the `disabled` state.

```jsx live manual compact
// Slider.tsx
// import * as SliderPrimitive from '@radix-ui/react-slider';

const Slider = styled(SliderPrimitive, {
  position: 'relative',
  display: 'flex',
  alignItems: 'center',
  userSelect: 'none',
  touchAction: 'none',
  height: 16,
});

const SliderTrack = styled(SliderPrimitive.Track, {
  backgroundColor: 'gainsboro',
  position: 'relative',
  flexGrow: 1,
  height: 2,
});

const SliderRange = styled(SliderPrimitive.Range, {
  position: 'absolute',
  backgroundColor: 'dodgerblue',
  borderRadius: '9999px',
  height: '100%',
  '[data-disabled] &': {
    backgroundColor: 'darkgray',
  },
});

const SliderThumb = styled(SliderPrimitive.Thumb, {
  display: 'block',
  width: 16,
  height: 16,
  backgroundColor: 'white',
  border: '1px solid lightgray',
  borderRadius: '20px',
  ':focus': {
    outline: 'none',
    borderColor: 'dodgerblue',
  },
});

render(
  <Slider defaultValue={50} disabled>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>
    <SliderThumb />
  </Slider>
);
```

`SliderTrack` is the track part.

#### Range

`SliderRange` is the range part. Must live inside `SliderTrack`.

#### Thumb

`SliderThumb` is the thumb part. You can render multiple thumbs.
